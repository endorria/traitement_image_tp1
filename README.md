/* ------------------------------------------------- */
void sort_selection_ui8vector(uint8 *X, int i0, int i1)
/* ------------------------------------------------- */
{
    int c,d;
    uint8 swap;
    for (c = i0; c < i1; c++) {
        for (d = i0 ; d < i1 - c; d++) {
            if (X[d] > X[d+1]){
            /* For decreasing order use < */
                swap = X[d];
                X[d] = X[d+1];
                X[d+1] = swap;
            }
        }
    }
}
